    Attributes
        ID   : 2

    Events
        Touch Press Event
            swipestore.en=1 // Start swipestore timer
        
        Touch Release Event
            swipestore.en=0
            // Touch has ended, x
            if(tch0==0)
            {
                swipec=swipex-tch2
                // From Left to Right
                if(swipec>swipedx&&api==1)
                {
                    page buttonpage04
                }
                // Right to Left
                swipec2=0-swipedx
                if(swipec<swipec2&&api==1)
                {
                    page buttonpage01
                }
            }
            // Touch has ended, y
            if(tch1==0)
            {
                swipec=swipey-tch3
                // From Up to Down
                if(swipec>100&&api==1)
                {
                    page buttonpage02
                }
                // Down to Up
                swipec2=0-swipedy
                if(swipec<-100&&api==1)
                {
                    page buttonpage03
                }
            }
