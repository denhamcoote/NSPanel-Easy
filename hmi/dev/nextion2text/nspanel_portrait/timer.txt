            // Decrement
            if(remaining.val>0)
            {
                remaining.val--
            }
            // Select font and build string based on remaining seconds
            if(remaining.val>=3600)
            {
                countdown.font=6
                sys0=remaining.val/3600
                sys2=remaining.val%3600
                sys1=sys2/60
                sys2=sys2%60
                covx sys0,va0.txt,0,0
                countdown.txt=va0.txt+time_sep.txt
                if(sys1<10)
                {
                    countdown.txt+="0"
                }
                covx sys1,va0.txt,0,0
                countdown.txt+=va0.txt+time_sep.txt
                if(sys2<10)
                {
                    countdown.txt+="0"
                }
                covx sys2,va0.txt,0,0
                countdown.txt+=va0.txt
            }else
            {
                if(remaining.val>=60)
                {
                    countdown.font=11
                    sys1=remaining.val/60
                    sys2=remaining.val%60
                    covx sys1,va0.txt,0,0
                    countdown.txt=va0.txt+time_sep.txt
                    if(sys2<10)
                    {
                        countdown.txt+="0"
                    }
                    covx sys2,va0.txt,0,0
                    countdown.txt+=va0.txt
                }else
                {
                    countdown.font=12
                    covx remaining.val,countdown.txt,0,0
                }
            }
            // Stop timer when expired
            if(remaining.val==0)

                tick.en=0
